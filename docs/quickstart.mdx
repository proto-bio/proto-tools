# Quickstart

## Installation

```bash bash icon="terminal"
pip install bio-programming-tools
```

## Basic Usage

### Structure Prediction

```python python icon="python"
from bio_programming_tools.tools.structure_prediction import run_esmfold, ESMFoldInput, ESMFoldConfig

# Predict structure
input_data = ESMFoldInput(sequences=["MVLSPADKTN"])
config = ESMFoldConfig(device="cuda")
result = run_esmfold(input_data, config)

# Access results
structure = result.structures[0]
print(f"pLDDT: {structure.plddt}")
structure.save("output.pdb")
```

### Inverse Folding

```python python icon="python"
from bio_programming_tools.tools.inverse_folding import (
    run_proteinmpnn_sample,
    InverseFoldingInput,
    InverseFoldingConfig
)

# Design sequences for a structure
inputs = InverseFoldingInput(structures=["protein.pdb"])
config = InverseFoldingConfig(batch_size=10, temperature=0.1)
result = run_proteinmpnn_sample(inputs, config)

# Get designed sequences
for i, seq in enumerate(result[0].sequences):
    print(f"Design {i+1}: {seq[:50]}... (perplexity: {result[0].perplexity[i]:.2f})")
```

### Gene Annotation

```python python icon="python"
from bio_programming_tools.tools.gene_annotation import run_pyhmmer, PyHMMERInput, PyHMMERConfig
from bio_programming_tools.entities import Sequence

# Search with HMM profiles
query = Sequence(sequence="MVLSPADKTN", sequence_type="protein")
inputs = PyHMMERInput(
    query_sequences=[query],
    database_path="pfam.hmm"
)
config = PyHMMERConfig(evalue_threshold=1e-5)
result = run_pyhmmer(inputs, config)

# Access hits
for hit in result.hits:
    print(f"{hit.target_name}: E-value {hit.evalue}")
```

### Sequence Alignment

```python python icon="python"
from bio_programming_tools.tools.sequence_alignment import run_mafft, MAFFTInput, MAFFTConfig
from bio_programming_tools.entities import Sequence

# Align sequences
sequences = [
    Sequence(sequence="MVLSPADKTN", sequence_type="protein"),
    Sequence(sequence="MVLSAADKTN", sequence_type="protein")
]
inputs = MAFFTInput(sequences=sequences)
config = MAFFTConfig(algorithm="auto")
result = run_mafft(inputs, config)

# Get alignment
for aligned_seq in result.aligned_sequences:
    print(aligned_seq.sequence)
```

## Common Patterns

### Working with Structures

```python python icon="python"
from bio_programming_tools.entities import Structure

# Load from file
structure = Structure(structure_filepath_or_content="protein.pdb")

# Access structure data
print(f"Chains: {structure.get_chain_ids()}")
print(f"Length: {len(structure.get_sequence())}")

# Save to file
structure.save("output.pdb")
```

### GPU Configuration

```python python icon="python"
# Use specific GPU
config = ESMFoldConfig(device="cuda:1")

# CPU fallback
config = ESMFoldConfig(device="cpu")

# Let bio-programming-tools choose
config = ESMFoldConfig(device="cuda")  # Uses CUDA_VISIBLE_DEVICES
```

### Logging

```python python icon="python"
from bio_programming_tools import setup_logging
import logging

# Enable logging
setup_logging(level=logging.INFO)

# Get logger
from bio_programming_tools import get_logger
logger = get_logger(__name__)

logger.info("Running prediction...")
```
